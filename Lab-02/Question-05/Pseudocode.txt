START

DISPLAY menu
INPUT choice

SWITCH(choice)

CASE 1:
    INPUT voltage,current
    power = voltage * current
    DISPLAY power
    BREAK

CASE 2:
    INPUT voltage,current
    resistance = voltage/current
    DISPLAY resistance
    BREAK

CASE 3:
    INPUT power,time
    energy = power*time
    DISPLAY energy
    BREAK

DEFAULT:
    DISPLAY "Invalid Choice"

END SWITCH

STOP
