BEGIN

DECLARE binaryNum AS LONG INTEGER
DECLARE decimalNum = 0
DECLARE i = 0

DISPLAY "Enter a binary number:"
INPUT binaryNum

WHILE binaryNum ≠ 0 DO

    remainder = binaryNum MOD 10

    decimalNum = decimalNum + (remainder × 2^i)

    binaryNum = binaryNum DIV 10

    i = i + 1

END WHILE

DISPLAY "Decimal equivalent =", decimalNum

END
